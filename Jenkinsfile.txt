pipeline {
    agent any

    stages {
        stage ('Compile Stage') {

            steps {
                withMaven(maven : 'maven_3_9_2') {
                    bat 'mvn.cmd clean compile'
                }
            }
        }

        stage ('Package Stage') {
            steps {
                withMaven(maven : 'maven_3_9_2') {
                    bat 'mvn.cmd package'
                }
            }
        }
    }
}
